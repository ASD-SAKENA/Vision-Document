
\فصل{نیازمندی‌های اولیه}

\قسمت{نیازمندی‌های عملکردی (\lr{Functional Requirements})}

\زیرقسمت{احراز هویت و مدیریت حساب کاربری}

\شروع{شمارش}
\فقره سیستم باید امکان ثبت‌نام کاربران جدید (ساکن، کارکن خدماتی) را فراهم کند.
\فقره سیستم باید امکان ورود و خروج کاربران را با احراز هویت ایمن فراهم کند.
\فقره سیستم باید دسترسی به قابلیت‌های تعاملی را برای کاربران احراز هویت‌نشده محدود کند.
\فقره سیستم باید کاربران را بر اساس نقش‌شان (ساکن، کارکن خدماتی، مدیر ساختمان، مدیر سامانه) تفکیک کند.
\فقره کاربران باید بتوانند پروفایل خود را ویرایش کنند.
\فقره سیستم باید امکان بازیابی رمز عبور را فراهم کند.
\فقره مدیر سامانه باید بتواند حساب‌های مشکل‌دار را غیرفعال کند.
\پایان{شمارش}

\زیرقسمت{مدیریت ساختمان و واحدها}

\شروع{شمارش}
\فقره مدیر ساختمان باید بتواند اطلاعات ساختمان و واحدها را ثبت و ویرایش کند.
\فقره مدیر باید بتواند ساکنین را به واحدها تخصیص دهد.
\فقره ساکن باید بتواند اطلاعات واحد مسکونی خود را مشاهده کند.
\پایان{شمارش}

\زیرقسمت{درخواست‌های خدماتی}

\شروع{شمارش}
\فقره ساکنین باید بتوانند درخواست خدمات یا تعمیرات ثبت کنند.
\فقره ساکنین باید بتوانند وضعیت درخواست‌های خود را پیگیری کنند.
\فقره مدیر باید بتواند درخواست‌های ثبت‌شده را مشاهده و تأیید کند.
\فقره مدیر باید بتواند درخواست را به کارکن خدماتی ارجاع دهد.
\فقره کارکن خدماتی باید درخواست‌های ارجاع‌شده را مشاهده کند.
\فقره کارکن باید بتواند گزارش انجام کار و وضعیت نهایی درخواست را ثبت کند.
\پایان{شمارش}

\زیرقسمت{شارژ و پرداخت}

\شروع{شمارش}
\فقره مدیر باید بتواند دوره‌های شارژ و هزینه‌های اضافی را تعریف کند.
\فقره مدیر باید بتواند ضریب هزینه را برای هر واحد تعیین کند.
\فقره ساکنین باید بتوانند از طریق کیف پول داخلی شارژ پرداخت کنند.
\فقره ساکنین باید بتوانند تاریخچه‌ی پرداخت‌های خود را مشاهده کنند.
\فقره مدیر باید بتواند وضعیت پرداخت‌ها و بدهکاران را مشاهده کند.
\پایان{شمارش}

\زیرقسمت{رزرو امکانات مشترک}

\شروع{شمارش}
\فقره مدیر باید بتواند امکانات قابل رزرو را تعریف، ویرایش و حذف کند.
\فقره مدیر باید بتواند ظرفیت و زمان‌بندی هر امکان را تعیین کند.
\فقره ساکنین باید بتوانند امکانات مشترک را آنلاین رزرو کنند.
\فقره سیستم باید از ثبت رزروهای متداخل جلوگیری کند.
\فقره ساکنین باید بتوانند رزرو خود را لغو کنند.
\فقره ساکنین باید بتوانند تاریخچه‌ی رزروهای خود را مشاهده کنند.
\پایان{شمارش}

\زیرقسمت{اطلاع‌رسانی و گزارش‌گیری}

\شروع{شمارش}
\فقره مدیر باید بتواند اطلاعیه برای ساکنین منتشر کند.
\فقره ساکنین باید بتوانند اطلاعیه‌ها و آرشیو آن‌ها را مشاهده کنند.
\فقره مدیر باید بتواند گزارش‌های مرتبط با پرداخت‌ها، درخواست‌ها و رزروها را دریافت کند.
\پایان{شمارش}

\قسمت{اولویت‌بندی نیازمندی‌ها}

بر اساس دسته‌بندی \lr{Lean DAD} برای آیتم‌های کاری:

\شروع{فقرات}
\فقره \textbf{\lr{Business Value}:} احراز هویت، مدیریت واحدها، درخواست تعمیرات، شارژ و پرداخت، رزرو امکانات، اطلاعیه‌ها
\فقره \textbf{\lr{Date Fixed}:} تحویل نسخه‌ی اول در پایان ایتریشن دوم (اواسط تیر ۱۴۰۵)
\فقره \textbf{\lr{Intangible}:} تنظیمات امنیتی، \lr{RBAC}، پشتیبان‌گیری، مستندات فنی
\فقره \textbf{\lr{Could Have}:} پیام‌رسانی داخلی، نظرسنجی، کیف پول با شارژ واقعی
\پایان{فقرات}
